288 static Vec6T Log(const SE3& x)
289 {
290 Vec3T w = SO3<T>::Log(x.q_);
291 T th = w.norm();
292 Mat3T W = SO3<T>::hat(w);
293
294 Mat3T leftJacobianInverse;
295 if (th > (T)1e-4)
296 {
297 T a = sin(th) / th;
298 T b = ((T)1. - cos(th)) / (th * th);
299 T c = ((T)1. - a) / (th * th);
300 T e = (b - 2. * c) / (2. * a);
301 leftJacobianInverse = Mat3T::Identity() - 0.5 * W + e * (W * W);
302 }
303 else
304 {
305 leftJacobianInverse = Mat3T::Identity();
306 }
307
308 Vec6T out;
309 out << leftJacobianInverse * x.t_, w;
310
311 return out;
312 }

319 static SE3 Exp(const Vec6T& omega)
320 {
321 Vec3T rho = omega.template block<3, 1>(0, 0);
322 Vec3T w = omega.template block<3, 1>(3, 0);
323 SO3<T> q = SO3<T>::Exp(w);
324 Mat3T W = SO3<T>::hat(w);
325 T th = w.norm();
326
327 Mat3T leftJacobian;
328 if (th > (T)1e-4)
329 {
330 T a = sin(th) / th;
331 T b = ((T)1. - cos(th)) / (th * th);
332 T c = ((T)1. - a) / (th * th);
333 leftJacobian = a * Mat3T::Identity() + b * W + c * (w * w.transpose());
334 }
335 else
336 {
337 leftJacobian = Mat3T::Identity();
338 }
339
340 return SE3::fromVecAndQuat(leftJacobian * rho, q);
341 }
