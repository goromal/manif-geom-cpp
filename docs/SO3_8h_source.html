70 static SO3 fromR(const Mat3T& m)
71 {
72 T s, qw, qx, qy, qz;
73
74 T R11 = m(0, 0);
75 T R12 = m(0, 1);
76 T R13 = m(0, 2);
77 T R21 = m(1, 0);
78 T R22 = m(1, 1);
79 T R23 = m(1, 2);
80 T R31 = m(2, 0);
81 T R32 = m(2, 1);
82 T R33 = m(2, 2);
83
84 if (R11 + R22 + R33 > (T)0.0)
85 {
86 s = 2. * sqrt(1. + R11 + R22 + R33);
87 qw = 0.25 * s;
88 qx = (R32 - R23) / s;
89 qy = (R13 - R31) / s;
90 qz = (R21 - R12) / s;
91 }
92 else if (R11 > R22 && R11 > R33)
93 {
94 s = 2. * sqrt(1. + R11 - R22 - R33);
95 qw = (R32 - R23) / s;
96 qx = 0.25 * s;
97 qy = (R21 + R12) / s;
98 qz = (R31 + R13) / s;
99 }
100 else if (R22 > R33)
101 {
102 s = 2. * sqrt(1. + R22 - R11 - R33);
103 qw = (R13 - R31) / s;
104 qx = (R21 + R12) / s;
105 qy = 0.25 * s;
106 qz = (R32 + R23) / s;
107 }
108 else
109 {
110 s = 2. * sqrt(1. + R33 - R11 - R22);
111 qw = (R21 - R12) / s;
112 qx = (R31 + R13) / s;
113 qy = (R32 + R23) / s;
114 qz = 0.25 * s;
115 }
116
117 SO3 q = SO3::fromQuat(qw, qx, qy, qz);
118 q.normalize();
119
120 return q;
121 }

123 static SO3 fromTwoUnitVectors(const Vec3T& u, const Vec3T& v)
124 {
125 SO3 q;
126 T d = u.dot(v);
127
128 if (d < (T)0.99999999 && d > (T)-0.99999999)
129 {
130 T invs = 1. / sqrt((2. * (1. + d)));
131 Vec3T xyz = u.cross(v * invs);
132 q = SO3::fromQuat(0.5 / invs, xyz(0), xyz(1), xyz(2));
133 q.normalize();
134 }
135 else if (d < (T)-0.99999999)
136 {
137 // There are an infinite number of solutions here, choose one.
138 // This choice works better for vector comparisons with only
139 // nonzero x components.
140 q = SO3::fromQuat((T)0., (T)0., (T)1., (T)0.);
141 }
142 else
143 {
144 q = SO3::identity();
145 }
146
147 return q;
148 }

461 static SO3 Exp(const Vec3T& omega)
462 {
463 T th = omega.norm();
464
465 SO3 q;
466 if (th > (T)1e-4)
467 {
468 Vec3T u = omega / th;
469 q.arr_(0) = cos(th / 2.0);
470 q.arr_.template block<3, 1>(1, 0) = sin(th / 2.0) * u;
471 }
472 else
473 {
474 q.arr_(0) = (T)1.0;
475 q.arr_.template block<3, 1>(1, 0) = omega / 2.0;
476 q.normalize();
477 }
