303 static Vec6T Log(const SE3& x)
304 {
305 Vec3T w = SO3<T>::Log(x.q_);
306 T th = w.norm();
307 Mat3T W = SO3<T>::hat(w);
308
309 Mat3T leftJacobianInverse;
310 if (th > (T)1e-4)
311 {
312 T a = sin(th) / th;
313 T b = ((T)1. - cos(th)) / (th * th);
314 T c = ((T)1. - a) / (th * th);
315 T e = (b - 2. * c) / (2. * a);
316 leftJacobianInverse = Mat3T::Identity() - 0.5 * W + e * (W * W);
317 }
318 else
319 {
320 leftJacobianInverse = Mat3T::Identity();
321 }
322
323 Vec6T out;
324 out << leftJacobianInverse * x.t_, w;
325
326 return out;
327 }

334 static SE3 Exp(const Vec6T& omega)
335 {
336 Vec3T rho = omega.template block<3, 1>(0, 0);
337 Vec3T w = omega.template block<3, 1>(3, 0);
338 SO3<T> q = SO3<T>::Exp(w);
339 Mat3T W = SO3<T>::hat(w);
340 T th = w.norm();
341
342 Mat3T leftJacobian;
343 if (th > (T)1e-4)
344 {
345 T a = sin(th) / th;
346 T b = ((T)1. - cos(th)) / (th * th);
347 T c = ((T)1. - a) / (th * th);
348 leftJacobian = a * Mat3T::Identity() + b * W + c * (w * w.transpose());
349 }
350 else
351 {
352 leftJacobian = Mat3T::Identity();
353 }
354
355 return SE3::fromVecAndQuat(leftJacobian * rho, q);
