300 static SE2 Exp(const Vec3T& omega)
301 {
302 Vec2T rho = omega.template block<2, 1>(0, 0);
303 Vec1T w = omega.template block<1, 1>(2, 0);
304 SO2<T> q = SO2<T>::Exp(w);
305 T th = w.norm();
306
307 Mat2T leftJacobian;
308 if (abs(th) > (T)1e-4)
309 {
310 T a = sin(th) / th;
311 T b = ((T)1. - cos(th)) / th;
312 Mat2T skew1;
313 skew1 << (T)0, (T)-1., (T)1., (T)0;
314 leftJacobian = a * Mat2T::Identity() + b * skew1;
315 }
316 else

318 leftJacobian = Mat2T::Identity();
319 }
320
